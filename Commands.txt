# 1. Creating and Renaming Files

mv example.txt renamed_example.txt

# 2. Viewing File Contents
cat /etc/passwd
head -n 5 /etc/passwd
tail -n 5 /etc/passwd

grep "root" /etc/passwd

cd ..
zip -r test_dir.zip test_dir
mkdir unzipped_dir
unzip test_dir.zip -d unzipped_dir
